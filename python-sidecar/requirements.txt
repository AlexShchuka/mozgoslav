# first sidecar start on Linux dev boxes. The CI Dockerfile

# --- ML runtime ---
# torch pulls ~4GB of CUDA wheels on Linux by default — overkill for the
# sidecar which runs CPU-only. Pin Linux to the CPU-only index; macOS wheels
# are CPU-native already and resolve from PyPI. PEP 440 forbids `>=` with
# local version labels (`+cpu`), so Linux uses exact-pin.
--extra-index-url https://download.pytorch.org/whl/cpu

torch==2.6.0+cpu ; sys_platform == 'linux'
